# Invalid and edge cases
SRC= SRC=10.0.0.2 PROTO=UDP SPT=1234 DPT=5678 ACTION=BLOCKED         # Duplicate SRC
SRC=256.300.1.1 DST=192.168.1.1 PROTO=TCP SPT=1024 DPT=80 ACTION=BLOCKED  # Invalid IP

PROTO=UDP ACTION=ALLOWED                                             # Missing SRC and DST
SRC=10.0.0.10 DST=10.0.0.20 PROTO=UDP                                # Missing ACTION

SRC=192.168.1.10 DST=10.0.0.2 PROTO=TCP SPT=0 DPT=0 ACTION=ALLOWED   # Zero ports
